  push:
    branches: [ source ]

  build:

        access-token: ${{ secrets.GATSBY_PUBLISH_ACCESS_TOKEN }}

        # Additional arguments that get passed to `gatsby build`.
        gatsby-args: gh-pages -d public # optional, default i
